├── config/                 # Thư mục cấu hình lõi hệ thống
│   └── database.php        # Thiết lập kết nối CSDL bằng PDO an toàn
├── css/                    # Tài nguyên định dạng (Cascading Style Sheets)
├── img/                    # Tài nguyên hình ảnh
├── js/                     # Tài nguyên kịch bản (JavaScript)

│   ├── index.php           # Bộ định tuyến của Giảng viên
│   ├── grades.php          # Module xem danh sách lớp học phần
│   └── grade_detail.php    # Module nhập điểm chi tiết
├── student/                # Phân hệ Sinh viên (Student)
│   ├── includes/           # Giao diện dùng chung của Sinh viên
│   ├── index.php           # Bộ định tuyến của Sinh viên
│   ├── course_classes.php  # Module xử lý Đăng ký học phần
│   ├── timetable.php       # Module xem Thời khóa biểu cá nhân
│   └── grades.php          # Module xem Kết quả học tập
├── vendor/                 # Thư viện mã nguồn bên thứ ba (Third-party)
├── index.php               # Cổng điều hướng gốc của toàn bộ ứng dụng
